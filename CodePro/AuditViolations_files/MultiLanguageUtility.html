16 /**
17  * Utility for supporting multiple languages in the view.
18  * The utility works by reading a list of resource files for different languages and
19  * uses JavaFX ObjectProperties (which use native JavaFX Observer/Observable pattern)
20  * to bind JavaFX nodes to certain texts. Furthermore, switching between languages is
21  * supported by a call to the setLanguage(String languageName) method. For any JavaFX
22  * node to update it's text it requires a 1-time setup to bind the property
23  * e.g.,
24  * myButton.getTextProperty().bind(MultiLanguageUtility.newInstance().getString("BUTTON_LABEL"));
25  * 
26  * public API:
27  * getInstance() --> instance of the utility
28  * initLanguages(String ... resourceFiles) --> initializes the languages in the utility
29  * setLanguage(String language) --> sets the language of the utility
30  * getCurrentLanguage() --> returns the current language
31  * getSupportedLanguages() --> ObjectProperty holding ObservableList for supported languages
32  * getStringProperty(String propertyName) --> gets the string property (ObjectProperty holding
33  * String)
34  * getStringListProperty(String propertyName) --> gets the string list property (ObjectProperty
35  * holding ObservableList)
36  * Requires a resource (.properties) file for each language.
37  * The name of the language must be specified in the resource file as a tag "language=..."
38  * If it is not, then the name of the language is assumed to be the name of the file.
39  * 
40  * Singleton pattern: 1 MultiLanguageUtility instance per program to handle language
41  * (assumes that a program can only have 1 language at any given time)
42  * 
43  * @author Jonathan Tseng
44  *
45  */

192             String languageName =
193                     (bundle.keySet().contains(LANGUAGE_KEY)) ? bundle.getString(LANGUAGE_KEY)
194                                                             : fileName(file);
195             addLanguage(bundle, languageName);
196         });

209         bundle.keySet().forEach( (key) -> {

211                     new ArrayList<>(Arrays.asList(bundle.getString(key)
212                             .split(LIST_SPLITTER)));
213             if (values.size() == 1) {
214                 stringsMap.put(key, bundle.getString(key));
215             }
216                                 else {
217                                     listsMap.put(key, values);
218                                 }
219                             });
220         myLanguagesToStringsMap.put(languageName, stringsMap);
221         myLanguagesToStringListsMap.put(languageName, listsMap);
