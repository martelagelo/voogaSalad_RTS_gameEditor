219  private DIRECTION getDirection (Point2D loc, Point2D destination) {
220  double angle =
221  Math.atan2(-(destination.getY() - loc.getY()), destination.getX() - loc.getX());
222  double pi = Math.PI;
223  if ((-pi / 4.) < angle && angle <= (pi / 4.)) {
224  return DIRECTION.RIGHT;
225  }
226  else if ((pi / 8) < angle && angle <= (3 * pi / 8)) {
227  return DIRECTION.BK_RIGHT;
228  }
229  else if (3 * pi / 8 < angle && angle <= 5 * pi / 8) {
230  return DIRECTION.BK;
231  }
232  else if (5 * pi / 8 < angle && angle < 7 * pi / 8) {
233  return DIRECTION.BK_LEFT;
234  }
235  else if (7 * pi / 8 <= Math.abs(angle)) {
236  return DIRECTION.LEFT;
237  }
238  else if (-7 * pi / 8 < angle && angle <= -5 * pi / 8) {
239  return DIRECTION.FWD_LEFT;
240  }
241  else if (-5 * pi / 8 < angle && angle <= -3 * pi / 8) {
242  return DIRECTION.FWD;
243  }
244  else {
245  return DIRECTION.FWD_RIGHT;
