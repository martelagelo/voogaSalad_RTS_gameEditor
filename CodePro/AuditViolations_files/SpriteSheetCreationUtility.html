356         spritesheetGraphics.setColor(maskColor);
357         spritesheetGraphics.fillRect(0, 0, spritesheet.getWidth(), spritesheet.getHeight());
358
359         int currentColumn = 0;
360         int numRowsInCurrentColumn = 0;
361         for (int i = 0; i < images.size(); i++) {
362             BufferedImage image = images.get(i);
363             Location location = locations[i / numRows][i % numRows];
364             if ((i / numRows) != currentColumn) {
365                 currentColumn = i / numRows;
366                 rowCountsByColumn.add(numRowsInCurrentColumn);
367                 numRowsInCurrentColumn = 0;
368             }
369             numRowsInCurrentColumn++;
370             if (i == images.size() - 1) {
371                 rowCountsByColumn.add(numRowsInCurrentColumn);
372             }
373             spritesheetGraphics.drawImage(image, new Double(location.y - image.getHeight() / 2.0)
374                     .intValue(), new Double(location.x - image.getWidth() / 2.0).intValue(), null);
375         }
376         return spritesheet;
377     }

379     private BufferedImage tileSpritesheets (List<BufferedImage> spritesheets,
380                                             int frameWidth,
381                                             int frameHeight) {
382         int width = 0;
383         int height = 0;
384         for (BufferedImage spritesheet : spritesheets) {
385             width += spritesheet.getWidth();
386             if (spritesheet.getHeight() > height) {
387                 height = spritesheet.getHeight();
388             }
389         }
390
391         int type = BufferedImage.TYPE_INT_ARGB;
392         BufferedImage finalSpritesheet = new BufferedImage(width, height, type);
393         Graphics2D spritesheetGraphics = finalSpritesheet.createGraphics();
394         int currentWidth = 0;
395         for (BufferedImage spritesheet : spritesheets) {
396             spritesheetGraphics.drawImage(spritesheet, currentWidth, 0, null);
397             currentWidth += spritesheet.getWidth();
398         }
399
400         return finalSpritesheet;
401     }

462      *

464      *        image
465      * @return - image with all pink (0xFFFF00FF) pixels now transparent
466      * @throws IOException
467      */
468     private BufferedImage colorToTransparency (BufferedImage input, Color maskColor)
469                                                                                     throws IOException {

471             public final int filterRGB (int x, int y, int rgb) {

477         ImageProducer ip = new FilteredImageSource(input.getSource(), filter);
